DirectionalLight

A directional light is a type of light source that emits light in a specific direction from an infinitely far away source. It simulates light that is coming from a distant source, such as the sun. Unlike point lights or spotlights, directional lights have no position and emit parallel light rays.

Three.js DirectionalLight Example

    

        // Import necessary Three.js modules
        import * as THREE from 'three';
        import './style.css';  
        import * as dat from 'dat.gui';
        import {{'{'}} OrbitControls {{'}'}} from 'three/examples/jsm/controls/OrbitControls.js';

        // Three.js components
        let scene, camera, renderer, cube, sphere, torusKnot, ambientLight, directionalLight, controls, gui;

        // Initialize Three.js
        function initThree() {{'{'}}
          scene = new THREE.Scene();
          camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
          camera.position.z = 5;
          renderer = new THREE.WebGLRenderer({{'{'}} antialias: true {{'}'}});
          renderer.setSize(window.innerWidth, window.innerHeight);
          document.body.appendChild(renderer.domElement);

          window.addEventListener('resize', onWindowResize);
          {{'}'}}

        // Create scene elements
        function createScene() {{'{'}}
          ambientLight = new THREE.AmbientLight(0xffffff, 0.5);
          scene.add(ambientLight);
          {{'}'}}

        // Create lights
        function createLights() {{'{'}}
          directionalLight = new THREE.DirectionalLight(0xffffff, 1);
          directionalLight.position.set(1, 1, 1).normalize();
          scene.add(directionalLight);
          {{'}'}}

        // Create 3D objects
        function createObjects() {{'{'}}
          const cubeGeometry = new THREE.BoxGeometry();
          const cubeMaterial = new THREE.MeshStandardMaterial({{'{'}} color: 0x00ff00, metalness: 0.5, roughness: 0.5 {{'}'}});
          cube = new THREE.Mesh(cubeGeometry, cubeMaterial);
          scene.add(cube);

          const sphereGeometry = new THREE.SphereGeometry(1, 32, 32);
          const sphereMaterial = new THREE.MeshStandardMaterial({{'{'}} color: 0xff0000, metalness: 0.5, roughness: 0.5 {{'}'}});
          sphere = new THREE.Mesh(sphereGeometry, sphereMaterial);
          sphere.position.set(2, 0, 0);
          scene.add(sphere);

          const torusKnotGeometry = new THREE.TorusKnotGeometry(1, 0.4, 100, 16);
          const torusKnotMaterial = new THREE.MeshStandardMaterial({{'{'}} color: 0xff69b4, metalness: 0.5, roughness: 0.5 {{'}'}});
          torusKnot = new THREE.Mesh(torusKnotGeometry, torusKnotMaterial);
          torusKnot.position.set(-2, 0, 0);
          scene.add(torusKnot);
          {{'}'}}

        // Set up GUI controls
        function setupGui() {{'{'}}
          gui = new dat.GUI();

          const ambientLightColor = {{'{'}} color: ambientLight.color.getHex() {{'}'}};
          const directionalLightColor = {{'{'}} color: directionalLight.color.getHex() {{'}'}};
          const cubeColor = {{'{'}} color: cube.material.color.getHex() {{'}'}};
          const sphereColor = {{'{'}} color: sphere.material.color.getHex() {{'}'}};
          const torusKnotColor = {{'{'}} color: torusKnot.material.color.getHex() {{'}'}};

          const lightFolder = gui.addFolder('Lights');
          lightFolder.addColor(ambientLightColor, 'color').onChange(() => {{'{'}}
            ambientLight.color.set(ambientLightColor.color);
            {{'}'}});
          lightFolder.addColor(directionalLightColor, 'color').onChange(() => {{'{'}}
            directionalLight.color.set(directionalLightColor.color);
            {{'}'}});

          const cubeFolder = gui.addFolder('Cube Material');
          cubeFolder.addColor(cubeColor, 'color').onChange(() => {{'{'}}
            cube.material.color.set(cubeColor.color);
            {{'}'}});

          const sphereFolder = gui.addFolder('Sphere Material');
          sphereFolder.addColor(sphereColor, 'color').onChange(() => {{'{'}}
            sphere.material.color.set(sphereColor.color);
            {{'}'}});

          const torusKnotFolder = gui.addFolder('TorusKnot Material');
          torusKnotFolder.addColor(torusKnotColor, 'color').onChange(() => {{'{'}}
            torusKnot.material.color.set(torusKnotColor.color);
            {{'}'}});
            {{'}'}}

        // Set up orbit controls
        function setupOrbitControls() {{'{'}}
          controls = new OrbitControls(camera, renderer.domElement);
          controls.enableDamping = true;
          controls.dampingFactor = 0.25;
          controls.screenSpacePanning = false;
          controls.maxPolarAngle = Math.PI / 2;
          {{'}'}}

        // Animation loop
        function animate() {{'{'}}
          requestAnimationFrame(animate);

          // Rotate the objects
          cube.rotation.x += 0.01;
          cube.rotation.y += 0.01;
          sphere.rotation.x += 0.01;
          sphere.rotation.y += 0.01;
          torusKnot.rotation.x += 0.01;
          torusKnot.rotation.y += 0.01;

          // Update controls
          controls.update();

          // Render the scene
          renderer.render(scene, camera);
          {{'}'}}

        // Handle window resize
        function onWindowResize() {{'{'}}
          camera.aspect = window.innerWidth / window.innerHeight;
          camera.updateProjectionMatrix();
          renderer.setSize(window.innerWidth, window.innerHeight);
          {{'}'}}

        // Initialize Three.js components and start the animation loop
        initThree();
        createScene();
        createLights();
        createObjects();
        setupGui();
        setupOrbitControls();
        animate();


Explanation

Importing Necessary Libraries:

  • import * as THREE from 'three';: Imports the Three.js library, which is used for 3D graphics.
  • import './style.css';: Imports a CSS file for styling. The content of this file is not provided in the code snippet.
  • import * as dat from 'dat.gui';: Imports the dat.GUI library for creating a graphical user interface to control parameters.

Setting Up the Scene:

  • const scene = new THREE.Scene();: Creates a Three.js scene to hold 3D objects.
  • const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);: Creates a perspective camera with specified parameters.
  • const renderer = new THREE.WebGLRenderer();: Creates a WebGL renderer for rendering the 3D scene.
  • renderer.setSize(window.innerWidth, window.innerHeight);: Sets the renderer size to match the window size.
  • document.body.appendChild(renderer.domElement);: Appends the renderer's canvas element to the document body.

Creating Ambient and Directional Lights:

  • const directionalLight = new THREE.DirectionalLight(0xffffff, 1);: Creates a directional light with white color and intensity of 1.
  • scene.add(ambientLight); and scene.add(directionalLight);: Adds lights to the scene.

Creating 3D Objects:

  • const cubeMaterial = new THREE.MeshStandardMaterial({{'{'}} color: 0x00ff00, metalness: 0.5, roughness: 0.5 {{'}'}});: Creates a standard material for the cube with green color, metalness, and roughness properties.
  • const cube = new THREE.Mesh(cubeGeometry, cubeMaterial);: Creates a mesh (cube) with the specified geometry and material.
  • scene.add(cube);: Adds the cube to the scene.

GUI Controls for Lights and Material:

  • lightFolder.addColor(ambientLightColor, 'color').onChange(function () {{'{'}} ambientLight.color.set(ambientLightColor.color); {{'}'}});: Adds a color control for ambient light.
  • cubeFolder.addColor(cubeColor, 'color').onChange(function () {{'{'}} cube.material.color.set(cubeColor.color); {{'}'}});: Adds a color control for the cube's material.

Animation Loop and Window Resize Handling:

  • const animate = () => {{'{'}} ... {{'}'}};: Defines an animation loop using the requestAnimationFrame function.
  • window.addEventListener('resize', () => {{'{'}}... {{'}'}});: Event listener for window resize updates the camera aspect ratio and renderer size.

Starting the Animation:
